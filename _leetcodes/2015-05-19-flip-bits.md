可以直接逐位统计各个位，也可以异或后统计1的个数

	class Solution {
	public:
	    /**
	     *@param a, b: Two integer
	     *return: An integer
	     */
	    int bitSwapRequired(int a, int b) {
	        // write your code here
	        int result = a^b;
	        int counter = 0;
	        while (result) {
	            result = result & (result - 1);
	            counter++;
	        }
	        return counter;
	    }
	};

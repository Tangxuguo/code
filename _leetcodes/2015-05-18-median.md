比较老土的方法，排序后找

	class Solution {
	public:
	    /**
	     * @param nums: A list of integers.
	     * @return: An integer denotes the middle number of the array.
	     */
	    int median(vector<int> &nums) {
	        // write your code here
	        vector<int> tmp(nums);
	        sort(tmp.begin(), tmp.end());
	        if (tmp.size()%2)
	            return tmp[tmp.size()/2];
	        else
	            return tmp[tmp.size()/2-1];
	    }
	};
